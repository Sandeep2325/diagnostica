    function throttle(fn, wait) {

        return function () {
            if ((time + wait - Date.now()) < 0) {

            }

        });

        jQuery('.slider-wrp li').css({'width': max});

            sliderWidth += jQuery(elem).outerWidth(true);
            if (index > 0) {
